CREATE TABLE public.persona (
	id int NOT NULL GENERATED BY DEFAULT AS IDENTITY,
	nombre varchar NULL,
	apellidos varchar NULL,
	CONSTRAINT persona_pk PRIMARY KEY (id)
);
